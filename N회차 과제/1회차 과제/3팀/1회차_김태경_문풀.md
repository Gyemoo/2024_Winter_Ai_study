문제 출제자 - 박광석
======================

    import numpy as n

    x = [1, 2, 5, 8, 10, 12]
    y = [31, 62, 71, 76, 81, 96]
    
    xav = n.mean(x)
    yav = n.mean(y)
    
    def top(x, xav, y, yav):
      k = 0
      for i in range(len(x)):
        k += (x[i] - xav) * (y[i] - yav)
      return k
    
    calc1 = sum([(i - xav)**2 for i in x])
    calc2 = top(x, xav, y, yav)
    
    a = calc2 / calc1
    b = yav - (xav * a)
    
    X = int(input())
    Y = a*X + b
    
    print(Y)

문제 출제자 - 박경준
=========================

    import numpy as n
    import pandas as p
    import matplotlib.pyplot as mp
    
    data = [[3, 78], [5, 85], [7, 89], [9, 94]]
    x = [i[0] for i in data]
    y = [i[1] for i in data]
    
    xdata = n.array(x)
    ydata = n.array(y)
    
    a = 0
    b = 0
    
    learn = 0.02
    epoch = 3001
    
    for i in range(epoch):
      Y = a * xdata + b
      err = ydata - Y
      difa = -(2/len(xdata)) * sum(xdata * (err))
      difb = -(2/len(xdata)) * sum(err)
    
      a = a - learn * difa
      b = b - learn * difb
    
    Y = a * xdata + b
    mp.scatter(x, y)
    mp.plot([min(xdata), max(xdata)], [min(Y), max(Y)])
    mp.show()
